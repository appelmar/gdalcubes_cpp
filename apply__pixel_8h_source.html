69  apply_pixel_cube(std::shared_ptr<cube> in, std::vector<std::string> expr, std::vector<std::string> band_names = {}, bool keep_bands = false) : cube(in->st_reference()->copy()), _in_cube(in), _expr(expr), _band_names(band_names), _band_usage(), _band_usage_all(), _keep_bands(keep_bands) { // it is important to duplicate st reference here, otherwise changes will affect input cube as well
70  _chunk_size[0] = _in_cube->chunk_size()[0];
71  _chunk_size[1] = _in_cube->chunk_size()[1];
72  _chunk_size[2] = _in_cube->chunk_size()[2];
73 
74  if (_expr.empty()) {
75  GCBS_ERROR("No expression given");
76  throw std::string("ERROR in apply_pixel_cube::apply_pixel_cube(): No expression given");
77  }
78  if (!_band_names.empty()) {
79  if (_band_names.size() != _expr.size()) {
80  GCBS_ERROR("Numbers of provided expressions and band names differ");
81  throw std::string("ERROR in apply_pixel_cube::apply_pixel_cube(): Numbers of provided expressions and band names differ");
82  }
83  }
84 
85  if (_keep_bands) {
86  // TODO: check for band name conflicts here
87  for (uint16_t i = 0; i < _in_cube->size_bands(); ++i) {
88  _bands.add(_in_cube->bands().get(i));
89  }
90  }
91  for (uint16_t i = 0; i < _expr.size(); ++i) {
92  band b(_band_names.empty() ? ("x" + std::to_string(i + 1)) : _band_names[i]);
93  b.unit = "";
94  b.no_data_value = "nan";
95  b.type = "float64";
96  b.offset = 0;
97  b.scale = 1;
98  _bands.add(b);
99  }
100 
101  // tinyexpr works with lower case symbols only
102  for (uint16_t i = 0; i < _expr.size(); ++i) {
103  std::transform(_expr[i].begin(), _expr[i].end(), _expr[i].begin(), ::tolower);
104  }
105 
106  // parse expressions, currently this is only for validation,
107  // expressions will be parsed again in read_chunk(), costs should
108  // be negligible compared to the actual evaluation
109  if (!parse_expressions()) {
110  GCBS_ERROR("Invalid expression(s)");
111  throw std::string("ERROR in apply_pixel_cube::apply_pixel_cube(): Invalid expression(s)");
112  }
113 
114  // Find out, which bands are actually used per expression
115  for (uint16_t i = 0; i < _expr.size(); ++i) {
116  _band_usage.push_back(std::unordered_set<std::string>());
117  for (uint16_t ib = 0; ib < _in_cube->bands().count(); ++ib) {
118  std::string name = _in_cube->bands().get(ib).name;
119  std::string temp_name = name;
120 
121  // tinyexpr works with lower case symbols only
122  std::transform(temp_name.begin(), temp_name.end(), temp_name.begin(), ::tolower);
123 
124  if (_expr[i].find(temp_name) != std::string::npos) {
125  _band_usage[i].insert(name);
126  _band_usage_all.insert(name);
127  }
128  }
129  _var_usage.push_back(std::unordered_set<std::string>());
130  if (_expr[i].find("ix") != std::string::npos) _var_usage[i].insert("ix");
131  if (_expr[i].find("iy") != std::string::npos) _var_usage[i].insert("iy");
132  if (_expr[i].find("it") != std::string::npos) _var_usage[i].insert("it");
133  if (_expr[i].find("left") != std::string::npos) _var_usage[i].insert("left");
134  if (_expr[i].find("right") != std::string::npos) _var_usage[i].insert("right");
135  if (_expr[i].find("top") != std::string::npos) _var_usage[i].insert("top");
136  if (_expr[i].find("bottom") != std::string::npos) _var_usage[i].insert("bottom");
137  if (_expr[i].find("t0") != std::string::npos) _var_usage[i].insert("t0");
138  if (_expr[i].find("t1") != std::string::npos) _var_usage[i].insert("t1");
139  }
140  }
