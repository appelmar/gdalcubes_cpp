440  chunkid_t find_chunk_that_contains(coords_st p) const {
441  uint32_t cumprod = 1;
442  chunkid_t id = 0;
443 
444  // 1. Convert map coordinates to view-based coordinates
445  coords_nd<uint32_t, 3> s = _st_ref->cube_coords(p);
446 
447  // 2. Find out which chunk contains the integer view coordinates
448  id += cumprod * (s[2] / _chunk_size[2]);
449  cumprod *= (uint32_t)std::ceil(((double)_st_ref->nx()) / ((double)_chunk_size[2]));
450  id += cumprod * (s[1] / _chunk_size[1]);
451  cumprod *= (uint32_t)std::ceil(((double)(_st_ref->ny()) / ((double)_chunk_size[1])));
452  id += cumprod * (s[0] / _chunk_size[0]);
453  //cumprod *= (uint32_t)std::ceil(((double)(_c->view().nt()) / ((double)_size_t);
454 
455  return id;
456  }

463  bounds_nd<uint32_t, 3> chunk_limits(chunk_coordinate_tyx c) const {
464  cube_coordinate_tyx out_vcoords_low;
465  cube_coordinate_tyx out_vcoords_high;
466 
467  out_vcoords_low[0] = c[0] * _chunk_size[0];
468  out_vcoords_low[1] = c[1] * _chunk_size[1];
469  out_vcoords_low[2] = c[2] * _chunk_size[2];
470  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
471  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
472  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
473 
474  // Shrink to view
475  if (out_vcoords_high[0] >= _st_ref->nt())
476  out_vcoords_high[0] = _st_ref->nt() - 1;
477  if (out_vcoords_low[0] >= _st_ref->nt())
478  out_vcoords_low[0] = _st_ref->nt() - 1;
479 
480  if (out_vcoords_high[1] >= _st_ref->ny())
481  out_vcoords_high[1] = _st_ref->ny() - 1;
482  if (out_vcoords_low[1] >= _st_ref->ny())
483  out_vcoords_low[1] = _st_ref->ny() - 1;
484 
485  if (out_vcoords_high[2] >= _st_ref->nx())
486  out_vcoords_high[2] = _st_ref->nx() - 1;
487  if (out_vcoords_low[2] >= _st_ref->nx())
488  out_vcoords_low[2] = _st_ref->nx() - 1;
489 
490  bounds_nd<uint32_t, 3> out;
491  out.low = out_vcoords_low;
492  out.high = out_vcoords_high;
493  return out;
494  };

501  bounds_nd<uint32_t, 3> chunk_limits(chunkid_t id) const {
502  coords_nd<uint32_t, 3> out_vcoords_low;
503  coords_nd<uint32_t, 3> out_vcoords_high;
504 
505  // Transform to global coordinates based on chunk id
506  uint32_t cumprod = 1;
507  int32_t n;
508 
509  n = (uint32_t)std::ceil(((double)(_st_ref->nx())) / ((double)_chunk_size[2]));
510  out_vcoords_low[2] = (id / cumprod) % n;
511  out_vcoords_low[2] *= _chunk_size[2];
512  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
513  cumprod *= n;
514 
515  n = (uint32_t)std::ceil(((double)(_st_ref->ny())) / ((double)_chunk_size[1]));
516  out_vcoords_low[1] = (id / cumprod) % n;
517  out_vcoords_low[1] *= _chunk_size[1];
518  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
519  cumprod *= n;
520 
521  n = (uint32_t)std::ceil(((double)(_st_ref->nt())) / ((double)_chunk_size[0]));
522  out_vcoords_low[0] = (id / cumprod) % n;
523  out_vcoords_low[0] *= _chunk_size[0];
524  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
525  cumprod *= n;
526 
527  // Shrink to view
528  if (out_vcoords_high[0] >= _st_ref->nt())
529  out_vcoords_high[0] = _st_ref->nt() - 1;
530  if (out_vcoords_low[0] >= _st_ref->nt())
531  out_vcoords_low[0] = _st_ref->nt() - 1;
532 
533  if (out_vcoords_high[1] >= _st_ref->ny())
534  out_vcoords_high[1] = _st_ref->ny() - 1;
535  if (out_vcoords_low[1] >= _st_ref->ny())
536  out_vcoords_low[1] = _st_ref->ny() - 1;
537 
538  if (out_vcoords_high[2] >= _st_ref->nx())
539  out_vcoords_high[2] = _st_ref->nx() - 1;
540  if (out_vcoords_low[2] >= _st_ref->nx())
541  out_vcoords_low[2] = _st_ref->nx() - 1;
542 
543  bounds_nd<uint32_t, 3> out;
544  out.low = out_vcoords_low;
545  out.high = out_vcoords_high;
546  return out;
547  };
