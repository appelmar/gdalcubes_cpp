64  bounds_2d<Ta> transform(std::string srs_from, std::string srs_to) {
65  OGRSpatialReference srs_in;
66  OGRSpatialReference srs_out;
67  srs_in.SetFromUserInput(srs_from.c_str());
68  srs_out.SetFromUserInput(srs_to.c_str());
69 
70  if (srs_in.IsSame(&srs_out)) {
71  return *this;
72  }
73 
74  OGRCoordinateTransformation* coord_transform = OGRCreateCoordinateTransformation(&srs_in, &srs_out);
75 
76  Ta x[4] = {left, left, right, right};
77  Ta y[4] = {top, bottom, top, bottom};
78 
79  if (coord_transform == NULL || !coord_transform->Transform(4, x, y)) {
80  throw std::string("ERROR: coordinate transformation failed.");
81  }
82 
83  Ta xmin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
84  Ta ymin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
85  Ta xmax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
86  Ta ymax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
87  for (uint8_t k = 0; k < 4; ++k) {
88  if (x[k] < xmin) xmin = x[k];
89  if (y[k] < ymin) ymin = y[k];
90  if (x[k] > xmax) xmax = x[k];
91  if (y[k] > ymax) ymax = y[k];
92  }
93 
94  bounds_2d<Ta> in_extent;
95  left = xmin;
96  right = xmax;
97  top = ymax;
98  bottom = ymin;
99 
100  delete coord_transform;
101 
102  return *this;
103  }
