40  void remove(std::pair<uint32_t, uint32_t> key) {
41  _m.lock();
42  auto it = _cache.find(key);
43  if (it != _cache.end()) {
44  _size_bytes -= it->second->total_size_bytes();
45  _cache.erase(it);
46  }
47 
48  auto it_2 = _prio_forward.find(key);
49  uint32_t p = it_2->second;
50  if (it_2 != _prio_forward.end()) {
51  _prio_forward.erase(it_2);
52  }
53 
54  auto it_3 = _prio_backward.find(p);
55  if (it_3 != _prio_backward.end()) {
56  _prio_backward.erase(it_3);
57  }
58  _m.unlock();
59  }

61  void add(std::pair<uint32_t, uint32_t> key, std::shared_ptr<chunk_data> value) {
62  if (!has(key)) {
63  _m.lock();
64  while (_size_bytes + value->total_size_bytes() > config::instance()->get_server_chunkcache_max()) {
65  auto it = _prio_backward.lower_bound(0); // lowest value greater than or equal to 0
66  uint32_t p = it->first;
67 
68  // remove it from the data
69  remove(it->second);
70  }
71 
72  _cache[key] = value;
73  _size_bytes += value->total_size_bytes();
74 
75  uint64_t p = inc_prio(); // synchronize?
76 
77  _prio_forward[key] = p;
78  _prio_backward[p] = key;
79 
80  _size_bytes += value->total_size_bytes();
81  _m.unlock();
82  }
83  }

85  inline bool has(std::pair<uint32_t, uint32_t> key) {
86  return _cache.find(key) != _cache.end();
87  }

89  inline std::shared_ptr<chunk_data> get(std::pair<uint32_t, uint32_t> key) {
90  if (!has(key)) {
91  throw std::string("ERROR: in server_chunk_cache::get(): requested chunk is not available");
92  }
93  _m.lock();
94  uint64_t pnew = inc_prio();
95  auto it_1 = _prio_forward.find(key);
96  if (it_1 != _prio_forward.end()) {
97  uint32_t pold = it_1->second;
98  it_1->second = pnew;
99 
100  auto it_2 = _prio_backward.find(pold);
101  if (it_2 != _prio_backward.end()) {
102  _prio_backward.erase(it_2); // remove old
103  _prio_backward[pnew] = key; // add new
104  }
105  }
106  _m.unlock();
107  return _cache[key];
108  }

110  inline uint32_t total_size_bytes() {

114  private:
115  const uint64_t inc_prio() {
116  if (_cur_priority == std::numeric_limits<uint64_t>::max()) {
117  uint64_t newp = 0;
118  for (auto it = _prio_forward.begin(); it != _prio_forward.end(); ++it) {
119  it->second = newp++;
120  _prio_backward.insert(std::make_pair(it->second, it->first));
121  }
122  _cur_priority = newp;
123  } else {
124  ++_cur_priority;
125  }
126  return _cur_priority;
127  }
