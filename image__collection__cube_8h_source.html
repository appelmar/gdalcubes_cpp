40 struct value_mask : public image_mask {
41  public:
42  value_mask(std::unordered_set<double> mask_values, bool invert = false, std::vector<uint8_t> bits = std::vector<uint8_t>()) : _mask_values(mask_values), _invert(invert), _bits(bits) {}
43  void apply(double *mask_buf, double *pixel_buf, uint32_t nb, uint32_t ny, uint32_t nx) override {
44  uint32_t bitmask = 0;
45  if (!_bits.empty()) {
46  for (uint8_t ib = 0; ib < _bits.size(); ++ib) {
47  bitmask += (uint32_t)std::pow(2.0, (double)_bits[ib]);
48  }
49  }
50  if (!_invert) {
51  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
52  if (!_bits.empty()) {
53  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
54  }
55  if (_mask_values.count(mask_buf[ixy]) == 1) { // if mask has value
56  // set all bands to NAN
57  for (uint32_t ib = 0; ib < nb; ++ib) {
58  pixel_buf[ib * nx * ny + ixy] = NAN;
59  }
60  }
61  }
62  } else {
63  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
64  if (!_bits.empty()) {
65  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
66  }
67  if (_mask_values.count(mask_buf[ixy]) == 0) {
68  // set all bands to NAN
69  for (uint32_t ib = 0; ib < nb; ++ib) {
70  pixel_buf[ib * nx * ny + ixy] = NAN;
71  }
72  }
73  }
74  }
75  }
76 
77  json11::Json as_json() override {
78  json11::Json::object out;
79  out["mask_type"] = "value_mask";
80  out["values"] = _mask_values;
81  out["invert"] = _invert;
82  out["bits"] = _bits;
83  return out;
84  }

86  private:
87  std::unordered_set<double> _mask_values;
88  bool _invert;
89  std::vector<uint8_t> _bits;

92 struct range_mask : public image_mask {
93  public:
94  range_mask(double min, double max, bool invert = false, std::vector<uint8_t> bits = std::vector<uint8_t>()) : _min(min), _max(max), _invert(invert), _bits(bits) {}

96  void apply(double *mask_buf, double *pixel_buf, uint32_t nb, uint32_t ny, uint32_t nx) override {
97  uint32_t bitmask = 0;
98  if (!_bits.empty()) {
99  for (uint8_t ib = 0; ib < _bits.size(); ++ib) {
100  bitmask += (uint32_t)std::pow(2.0, (double)_bits[ib]);
101  }
102  }
103  if (!_invert) {
104  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
105  if (!_bits.empty()) {
106  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
107  }
108  if (mask_buf[ixy] >= _min && mask_buf[ixy] <= _max) {
109  // set all bands to NAN
110  for (uint32_t ib = 0; ib < nb; ++ib) {
111  pixel_buf[ib * nx * ny + ixy] = NAN;
112  }
113  }
114  }
115  } else {
116  for (uint32_t ixy = 0; ixy < ny * nx; ++ixy) {
117  if (!_bits.empty()) {
118  mask_buf[ixy] = (uint32_t)(mask_buf[ixy]) & bitmask;
119  }
120  if (mask_buf[ixy] < _min || mask_buf[ixy] > _max) {
121  // set all bands to NAN
122  for (uint32_t ib = 0; ib < nb; ++ib) {
123  pixel_buf[ib * nx * ny + ixy] = NAN;
124  }
125  }
126  }
127  }
128  }
