91  void add(std::pair<uint32_t, uint32_t> key, std::shared_ptr<chunk_data> value) {
92  if (!has(key)) {
93  _m.lock();
94  while (_size_bytes + value->total_size_bytes() > config::instance()->get_server_chunkcache_max()) {
95  auto it = _prio_backward.lower_bound(0); // lowest value greater than or equal to 0
96 
97  // remove it from the data
98  remove(it->second);
99  }
100 
101  _cache[key] = value;
102  _size_bytes += value->total_size_bytes();
103 
104  uint64_t p = inc_prio(); // synchronize?
105 
106  _prio_forward[key] = p;
107  _prio_backward[p] = key;
108 
109  _size_bytes += value->total_size_bytes();
110  _m.unlock();
111  }
112  }

128  std::shared_ptr<chunk_data> get(std::pair<uint32_t, uint32_t> key) {
129  if (!has(key)) {
130  throw std::string("ERROR: in server_chunk_cache::get(): requested chunk is not available");
131  }
132  _m.lock();
133  uint64_t pnew = inc_prio();
134  auto it_1 = _prio_forward.find(key);
135  if (it_1 != _prio_forward.end()) {
136  uint32_t pold = it_1->second;
137  it_1->second = pnew;
138 
139  auto it_2 = _prio_backward.find(pold);
140  if (it_2 != _prio_backward.end()) {
141  _prio_backward.erase(it_2); // remove old
142  _prio_backward[pnew] = key; // add new
143  }
144  }
145  _m.unlock();
146  return _cache[key];
147  }

158  const uint64_t inc_prio() {
159  if (_cur_priority == std::numeric_limits<uint64_t>::max()) {
160  uint64_t newp = 0;
161  for (auto it = _prio_forward.begin(); it != _prio_forward.end(); ++it) {
162  it->second = newp++;
163  _prio_backward.insert(std::make_pair(it->second, it->first));
164  }
165  _cur_priority = newp;
166  } else {
167  ++_cur_priority;
168  }
